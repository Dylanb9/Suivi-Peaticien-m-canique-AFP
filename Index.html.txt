function loadState(callback){
  if (!firebaseReady){
    LOADING = false;
    callback();
    return;
  }
  db.ref('mp-afp-state').once('value').then(snap => {
    const val = snap.val();
    if (val) {
      STATE = val;
      if (!STATE.apprentices) STATE.apprentices = [];
    }
    LOADING = false;
    callback();
  }).catch(e => {
    console.error('load error', e);
    LOADING = false;
    callback();
  });

  // Ecoute en temps réel des changements (autres utilisateurs)
  db.ref('mp-afp-state').on('value', snap => {
    const val = snap.val();
    if (val) {
      STATE = val;
      if (!STATE.apprentices) STATE.apprentices = [];
      if (!LOADING) render();
    }
  });
}

  });
}

/* ====== HELPERS UI ====== */
function el(tag, attrs, children){
  const e = document.createElement(tag);
  attrs = attrs || {};
  Object.keys(attrs).forEach(k => {
    if (k === 'style') e.style.cssText = attrs[k];
    else if (k.startsWith('on')) e.addEventListener(k.slice(2), attrs[k]);
    else e.setAttribute(k, attrs[k]);
  });
  (children||[]).forEach(c => {
    if (typeof c === 'string') e.appendChild(document.createTextNode(c));
    else if (c) e.appendChild(c);
  });
  return e;
}

function card(children, extraStyle){
  const c = el('div', {class:'card', style: extraStyle||''}, children);
  return c;
}

function button(label, onClick, variant){
  const b = el('button', { onclick: onClick }, [label]);
  if (variant === 'primary') b.classList.add('primary');
  return b;
}

function badge(text, bg, color){
  return el('span', {class:'badge', style:'background:'+bg+';color:'+color+';'}, [text]);
}

/* ====== RENDU PRINCIPAL ====== */
function render(){
  root.innerHTML = '';
  if (LOADING){
    root.appendChild(el('div', {style:'padding:2rem;text-align:center;color:var(--text-secondary);font-size:14px;'}, ['Chargement…']));
    return;
  }
  if (!firebaseReady){
    root.appendChild(card([
      el('h3', {style:'margin-bottom:8px;'}, ['Configuration requise']),
      el('p', {style:'font-size:14px;color:var(--text-secondary);'}, ['La connexion à la base de données Firebase n\'est pas configurée. Modifiez les valeurs FIREBASE_CONFIG en haut du fichier HTML avec les identifiants de votre projet Firebase. Voir le guide fourni.'])
    ]));
    return;
  }
  if (!SESSION){ renderLogin(); }
  else if (SESSION.role === 'trainer'){ renderTrainerDashboard(); }
  else { renderApprenticeView(); }
}

function renderLogin(){
  const wrap = el('div', {style:'max-width:420px;margin:0 auto;padding:1rem 0;'});
  wrap.appendChild(el('h1', {style:'text-align:center;margin-bottom:4px;'}, ['Suivi de formation']));
  wrap.appendChild(el('p', {style:'text-align:center;color:var(--text-secondary);font-size:14px;margin-bottom:1.5rem;'}, ['Praticien-ne en mécanique AFP']));

  const tabBar = el('div', {style:'display:flex;gap:8px;margin-bottom:1.25rem;'});
  const content = el('div', {});

  function renderApprenticeLogin(){
    content.innerHTML = '';
    const list = STATE.apprentices || [];
    const selectWrap = el('div', {style:'margin-bottom:1rem;'});
    let sel = null;
    if (list.length === 0){
      selectWrap.appendChild(el('p', {style:'font-size:13px;color:var(--text-secondary);'}, ['Aucun apprenti enregistré encore. Demandez à votre formateur de créer votre profil.']));
    } else {
      selectWrap.appendChild(el('label', {style:'font-size:13px;color:var(--text-secondary);display:block;margin-bottom:6px;'}, ['Choisissez votre nom']));
      sel = el('select', {id:'appr-select'});
      sel.appendChild(el('option', {value:''}, ['— Sélectionner —']));
      list.forEach(a => sel.appendChild(el('option', {value:a.id}, [a.name])));
      selectWrap.appendChild(sel);
    }
    content.appendChild(selectWrap);

    const codeWrap = el('div', {style:'margin-bottom:1rem;'});
    codeWrap.appendChild(el('label', {style:'font-size:13px;color:var(--text-secondary);display:block;margin-bottom:6px;'}, ['Votre code personnel']));
    const codeInput = el('input', {type:'password', id:'appr-code', placeholder:'Code à 4 chiffres'});
    codeWrap.appendChild(codeInput);
    content.appendChild(codeWrap);

    const errBox = el('p', {style:'font-size:13px;color:var(--danger-text);min-height:18px;'}, ['']);
    content.appendChild(errBox);

    content.appendChild(button('Se connecter', () => {
      const id = sel ? sel.value : null;
      const code = document.getElementById('appr-code').value.trim();
      const appr = list.find(a => a.id === id);
      if (!appr){ errBox.textContent = 'Sélectionnez votre nom.'; return; }
      if (appr.code !== code){ errBox.textContent = 'Code incorrect.'; return; }
      SESSION = { role: 'apprentice', apprenticeId: appr.id };
      render();
    }, 'primary'));
  }

  function renderTrainerLogin(){
    content.innerHTML = '';
    const codeWrap = el('div', {style:'margin-bottom:1rem;'});
    codeWrap.appendChild(el('label', {style:'font-size:13px;color:var(--text-secondary);display:block;margin-bottom:6px;'}, ['Code formateur']));
    const codeInput = el('input', {type:'password', id:'trainer-code', placeholder:'Code formateur'});
    codeWrap.appendChild(codeInput);
    content.appendChild(codeWrap);
    const errBox = el('p', {style:'font-size:13px;color:var(--danger-text);min-height:18px;'}, ['']);
    content.appendChild(errBox);
    content.appendChild(button('Accéder au tableau de bord formateur', () => {
      const code = document.getElementById('trainer-code').value.trim();
      const expected = STATE.trainerCode || '0000';
      if (code !== expected){ errBox.textContent = 'Code incorrect.'; return; }
      SESSION = { role: 'trainer' };
      render();
    }, 'primary'));
    content.appendChild(el('p', {style:'font-size:12px;color:var(--text-tertiary);margin-top:1rem;'}, [
      STATE.trainerCode ? 'Code formateur configuré.' : 'Premier accès : code par défaut 0000. Changez-le ensuite dans Paramètres.'
    ]));
  }

  const tabBtnA = button('Je suis apprenti-e', () => setMode('apprentice'));
  const tabBtnT = button('Je suis formateur', () => setMode('trainer'));
  tabBtnA.style.flex = '1';
  tabBtnT.style.flex = '1';
  tabBar.appendChild(tabBtnA);
  tabBar.appendChild(tabBtnT);

  function setMode(m){
    if (m === 'apprentice') {
      renderApprenticeLogin();
      tabBtnA.classList.add('primary'); tabBtnT.classList.remove('primary');
    } else {
      renderTrainerLogin();
      tabBtnT.classList.add('primary'); tabBtnA.classList.remove('primary');
    }
  }

  wrap.appendChild(tabBar);
  wrap.appendChild(card([content]));
  root.appendChild(wrap);
  setMode('apprentice');
}

function progressSummary(appr){
  const total = RESOURCES.length;
  const validated = RESOURCES.filter(r => appr.progress[r.id] && appr.progress[r.id].status === 'validated').length;
  const pending = RESOURCES.filter(r => appr.progress[r.id] && appr.progress[r.id].status === 'done_self').length;
  const inProgress = RESOURCES.filter(r => appr.progress[r.id] && appr.progress[r.id].status === 'progress').length;
  const pct = Math.round((validated/total)*100);

  return card([
    el('div', {style:'display:flex;justify-content:space-between;align-items:baseline;margin-bottom:8px;'}, [
      el('span', {style:'font-size:13px;color:var(--text-secondary);'}, ['Compétences validées par le formateur']),
      el('span', {style:'font-size:20px;font-weight:500;'}, [pct + '%'])
    ]),
    el('div', {style:'height:8px;background:var(--bg);border-radius:99px;overflow:hidden;margin-bottom:12px;border:0.5px solid var(--border);'}, [
      el('div', {style:'height:100%;width:' + pct + '%;background:#3b6d11;border-radius:99px;'})
    ]),
    el('div', {style:'display:grid;grid-template-columns:repeat(4,1fr);gap:8px;'}, [
      miniStat(total, 'Total'), miniStat(validated, 'Validées'), miniStat(pending, 'En attente'), miniStat(inProgress, 'En cours'),
    ])
  ]);
}

function miniStat(num, label){
  return el('div', {style:'background:var(--bg);border-radius:var(--radius-md);padding:10px;text-align:center;border:0.5px solid var(--border);'}, [
    el('div', {style:'font-size:18px;font-weight:500;'}, [String(num)]),
    el('div', {style:'font-size:11px;color:var(--text-secondary);'}, [label])
  ]);
}

function statusColor(status){
  return {
    todo: {bg:'var(--bg)', text:'var(--text-secondary)'},
    progress: {bg:'var(--warning-bg)', text:'var(--warning-text)'},
    done_self: {bg:'var(--warning-bg)', text:'var(--warning-text)'},
    validated: {bg:'var(--success-bg)', text:'var(--success-text)'},
    rejected: {bg:'var(--danger-bg)', text:'var(--danger-text)'}
  }[status];
}

function renderApprenticeView(){
  const appr = getApprentice(SESSION.apprenticeId);
  if (!appr){ SESSION = null; render(); return; }

  const wrap = el('div', {});
  const header = el('div', {style:'display:flex;justify-content:space-between;align-items:center;margin-bottom:1.25rem;flex-wrap:wrap;gap:8px;'});
  header.appendChild(el('div', {}, [
    el('h2', {}, [appr.name]),
    el('p', {style:'margin:2px 0 0;font-size:13px;color:var(--text-secondary);'}, ['Mon suivi de progression'])
  ]));
  header.appendChild(button('Se déconnecter', () => { SESSION = null; render(); }));
  wrap.appendChild(header);

  wrap.appendChild(progressSummary(appr));

  const filterBar = el('div', {style:'display:flex;gap:8px;margin:1.25rem 0 1rem;flex-wrap:wrap;'});
  let activeDomain = 'all';
  const domainBtns = {};
  function setDomain(d){
    activeDomain = d;
    Object.keys(domainBtns).forEach(k => {
      if (k===d) domainBtns[k].classList.add('primary'); else domainBtns[k].classList.remove('primary');
    });
    renderList();
  }
  const allBtn = button('Tout', () => setDomain('all'));
  domainBtns['all'] = allBtn;
  filterBar.appendChild(allBtn);
  DOMAIN_ORDER.forEach(d => {
    const found = RESOURCES.find(r => r.domain===d);
    if (!found) return;
    const b = button(found.domainLabel, () => setDomain(d));
    domainBtns[d] = b;
    filterBar.appendChild(b);
  });
  wrap.appendChild(filterBar);

  const listWrap = el('div', {id:'res-list', style:'display:flex;flex-direction:column;gap:8px;'});
  wrap.appendChild(listWrap);

  function renderList(){
    listWrap.innerHTML = '';
    const items = RESOURCES.filter(r => activeDomain==='all' || r.domain===activeDomain);
    let lastSub = null;
    items.forEach(r => {
      if (r.subId !== lastSub){
        listWrap.appendChild(el('div', {style:'font-size:13px;font-weight:500;color:var(--text-secondary);margin:8px 0 0;'}, [r.subId + ' — ' + r.subTitle]));
        lastSub = r.subId;
      }
      listWrap.appendChild(resourceRow(appr, r));
    });
  }
  renderList();
  root.appendChild(wrap);
}

function resourceRow(appr, r){
  if (!appr.progress[r.id]) appr.progress[r.id] = {status:'todo', note:'', trainerNote:'', updatedAt:null};
  const p = appr.progress[r.id];
  const sc = statusColor(p.status);
  const row = el('div', {style:'border:0.5px solid var(--border);border-radius:var(--radius-md);padding:10px 12px;'});

  const top = el('div', {style:'display:flex;justify-content:space-between;align-items:flex-start;gap:10px;'});
  top.appendChild(el('p', {style:'font-size:14px;flex:1;'}, [r.text]));
  top.appendChild(badge(STATUS_LABELS[p.status], sc.bg, sc.text));
  row.appendChild(top);

  if (p.trainerNote){
    row.appendChild(el('p', {style:'font-size:12px;color:var(--info-text);margin-top:6px;'}, ['Retour du formateur : ' + p.trainerNote]));
  }

  const controls = el('div', {style:'display:flex;gap:6px;margin-top:8px;flex-wrap:wrap;'});
  const opts = [['todo','À commencer'], ['progress','En cours'], ['done_self','Je la considère acquise']];
  opts.forEach(function(pair){
    const val = pair[0], label = pair[1];
    const b = button(label, () => {
      if (p.status === 'validated' || p.status === 'rejected'){
        if (!confirm('Cette ressource a déjà été évaluée par le formateur. Continuer ?')) return;
      }
      p.status = val;
      p.updatedAt = new Date().toISOString();
      saveState().then(() => renderApprenticeView());
    });
    if (p.status === val) b.classList.add('primary');
    controls.appendChild(b);
  });
  row.appendChild(controls);
  return row;
}

function renderTrainerDashboard(){
  const wrap = el('div', {});
  const header = el('div', {style:'display:flex;justify-content:space-between;align-items:center;margin-bottom:1.25rem;flex-wrap:wrap;gap:8px;'});
  header.appendChild(el('div', {}, [
    el('h2', {}, ['Tableau de bord formateur']),
    el('p', {style:'margin:2px 0 0;font-size:13px;color:var(--text-secondary);'}, [(STATE.apprentices||[]).length + ' apprenti(s) suivi(s)'])
  ]));
  const headerBtns = el('div', {style:'display:flex;gap:8px;'});
  headerBtns.appendChild(button('Paramètres', () => { TRAINER_VIEW = 'settings'; renderTrainerDashboard(); }));
  headerBtns.appendChild(button('Se déconnecter', () => { SESSION = null; render(); }));
  header.appendChild(headerBtns);
  wrap.appendChild(header);

  if (TRAINER_VIEW === 'settings'){
    wrap.appendChild(renderSettingsPanel());
    root.innerHTML = ''; root.appendChild(wrap); return;
  }
  if (TRAINER_SELECTED_ID){
    wrap.appendChild(renderApprenticeDetailForTrainer(TRAINER_SELECTED_ID));
    root.innerHTML = ''; root.appendChild(wrap); return;
  }

  const addBar = el('div', {style:'display:flex;gap:8px;margin-bottom:1rem;'});
  const nameInput = el('input', {type:'text', placeholder:'Nom du nouvel apprenti', id:'new-appr-name', style:'flex:1;'});
  addBar.appendChild(nameInput);
  addBar.appendChild(button('Ajouter un apprenti', () => {
    const name = nameInput.value.trim();
    if (!name) return;
    const code = String(Math.floor(1000 + Math.random()*9000));
    const newAppr = { id: uid(), name: name, code: code, progress: emptyProgressFor() };
    if (!STATE.apprentices) STATE.apprentices = [];
    STATE.apprentices.push(newAppr);
    saveState().then(() => {
      alert('Apprenti ajouté : ' + name + '\n Code personnel : ' + code + '\n\nCommuniquez ce code à l\'apprenti-e.');
      renderTrainerDashboard();
    });
  }, 'primary'));
  wrap.appendChild(addBar);

  const apprentices = STATE.apprentices || [];
  if (apprentices.length === 0){
    wrap.appendChild(card([el('p', {style:'font-size:14px;color:var(--text-secondary);'}, ['Aucun apprenti enregistré. Ajoutez-en un ci-dessus pour commencer.'])]));
  } else {
    const grid = el('div', {style:'display:flex;flex-direction:column;gap:8px;'});
    apprentices.forEach(a => {
      const total = RESOURCES.length;
      const validated = RESOURCES.filter(r => a.progress[r.id] && a.progress[r.id].status === 'validated').length;
      const pendingReview = RESOURCES.filter(r => a.progress[r.id] && a.progress[r.id].status === 'done_self').length;
      const pct = Math.round((validated/total)*100);
      const rowChildren = [
        el('div', {style:'flex:1;'}, [
          el('p', {style:'font-weight:500;font-size:15px;'}, [a.name]),
          el('p', {style:'font-size:12px;color:var(--text-secondary);margin-top:2px;'}, ['Code : ' + a.code + ' · ' + validated + '/' + total + ' validées' + (pendingReview>0?(' · ' + pendingReview + ' en attente'):'')])
        ])
      ];
      if (pendingReview > 0){
        rowChildren.push(badge(pendingReview + ' à valider', 'var(--warning-bg)', 'var(--warning-text)'));
      }
      rowChildren.push(el('span', {style:'font-size:18px;font-weight:500;min-width:48px;text-align:right;'}, [pct + '%']));
      const row = card([
        el('div', {style:'display:flex;justify-content:space-between;align-items:center;gap:12px;cursor:pointer;', onclick: () => { TRAINER_SELECTED_ID = a.id; renderTrainerDashboard(); }}, rowChildren)
      ]);
      grid.appendChild(row);
    });
    wrap.appendChild(grid);
  }
  root.innerHTML = ''; root.appendChild(wrap);
}

function renderSettingsPanel(){
  const wrap = el('div', {});
  wrap.appendChild(button('← Retour', () => { TRAINER_VIEW = null; renderTrainerDashboard(); }));
  const c = card([
    el('h3', {style:'margin-bottom:8px;'}, ['Code formateur']),
    el('p', {style:'font-size:13px;color:var(--text-secondary);margin-bottom:10px;'}, ['Changez le code utilisé pour accéder au tableau de bord formateur.']),
    el('input', {type:'text', id:'new-trainer-code', placeholder:'Nouveau code', style:'margin-bottom:10px;'}),
    button('Mettre à jour le code', () => {
      const v = document.getElementById('new-trainer-code').value.trim();
      if (!v) return;
      STATE.trainerCode = v;
      saveState().then(() => alert('Code formateur mis à jour.'));
    }, 'primary')
  ], 'margin-top:1rem;');
  wrap.appendChild(c);

  const apprRows = (STATE.apprentices||[]).map(a => el('div', {style:'display:flex;justify-content:space-between;align-items:center;padding:6px 0;'}, [
    el('span', {style:'font-size:13px;'}, [a.name + ' (code ' + a.code + ')']),
    button('Supprimer', () => {
      if (!confirm('Supprimer définitivement ' + a.name + ' et toute sa progression ?')) return;
      STATE.apprentices = STATE.apprentices.filter(x => x.id !== a.id);
      saveState().then(() => renderTrainerDashboard());
    })
  ]));
  const dangerCard = card([
    el('h3', {style:'margin-bottom:8px;'}, ['Gestion des apprentis']),
    el('div', {style:'display:flex;flex-direction:column;'}, apprRows)
  ], 'margin-top:1rem;');
  wrap.appendChild(dangerCard);
  return wrap;
}

function renderApprenticeDetailForTrainer(apprId){
  const appr = getApprentice(apprId);
  const wrap = el('div', {});
  wrap.appendChild(button('← Retour à la liste', () => { TRAINER_SELECTED_ID = null; renderTrainerDashboard(); }));

  const headerRow = el('div', {style:'margin:1rem 0;'}, [
    el('h3', {}, [appr.name]),
    el('p', {style:'font-size:12px;color:var(--text-secondary);margin-top:2px;'}, ['Code personnel : ' + appr.code])
  ]);
  wrap.appendChild(headerRow);
  wrap.appendChild(progressSummary(appr));

  const filterBar = el('div', {style:'display:flex;gap:8px;margin:1.25rem 0 1rem;flex-wrap:wrap;'});
  let activeFilter = 'review';
  const filterBtns = {};
  function setFilter(f){
    activeFilter = f;
    Object.keys(filterBtns).forEach(k => {
      if (k===f) filterBtns[k].classList.add('primary'); else filterBtns[k].classList.remove('primary');
    });
    renderTrainerList();
  }
  const filters = [['review','À valider'], ['all','Tout voir'], ['validated','Validées'], ['rejected','À revoir']];
  filters.forEach(function(pair){
    const key = pair[0], label = pair[1];
    const b = button(label, () => setFilter(key));
    filterBtns[key] = b;
    filterBar.appendChild(b);
  });
  wrap.appendChild(filterBar);

  const listWrap = el('div', {style:'display:flex;flex-direction:column;gap:8px;'});
  wrap.appendChild(listWrap);

  function renderTrainerList(){
    listWrap.innerHTML = '';
    let items;
    if (activeFilter === 'review') items = RESOURCES.filter(r => appr.progress[r.id] && appr.progress[r.id].status === 'done_self');
    else if (activeFilter === 'validated') items = RESOURCES.filter(r => appr.progress[r.id] && appr.progress[r.id].status === 'validated');
    else if (activeFilter === 'rejected') items = RESOURCES.filter(r => appr.progress[r.id] && appr.progress[r.id].status === 'rejected');
    else items = RESOURCES;

    if (items.length === 0){
      listWrap.appendChild(card([el('p', {style:'font-size:13px;color:var(--text-secondary);'}, ['Rien à afficher dans cette catégorie.'])]));
      return;
    }
    let lastSub = null;
    items.forEach(r => {
      if (r.subId !== lastSub){
        listWrap.appendChild(el('div', {style:'font-size:13px;font-weight:500;color:var(--text-secondary);margin:8px 0 0;'}, [r.subId + ' — ' + r.subTitle]));
        lastSub = r.subId;
      }
      listWrap.appendChild(trainerResourceRow(appr, r, renderTrainerList));
    });
  }
  renderTrainerList();
  return wrap;
}

function trainerResourceRow(appr, r, refresh){
  if (!appr.progress[r.id]) appr.progress[r.id] = {status:'todo', note:'', trainerNote:'', updatedAt:null};
  const p = appr.progress[r.id];
  const sc = statusColor(p.status);
  const row = el('div', {style:'border:0.5px solid var(--border);border-radius:var(--radius-md);padding:10px 12px;'});
  const top = el('div', {style:'display:flex;justify-content:space-between;align-items:flex-start;gap:10px;'});
  top.appendChild(el('p', {style:'font-size:14px;flex:1;'}, [r.text]));
  top.appendChild(badge(STATUS_LABELS[p.status], sc.bg, sc.text));
  row.appendChild(top);

  const noteInput = el('input', {type:'text', placeholder:"Commentaire pour l'apprenti (optionnel)", value: p.trainerNote||'', style:'margin-top:8px;font-size:13px;'});
  row.appendChild(noteInput);

  const controls = el('div', {style:'display:flex;gap:6px;margin-top:8px;flex-wrap:wrap;'});
  controls.appendChild(el('button', {
    style:'border-color:#97c459;background:var(--success-bg);color:var(--success-text);font-weight:500;font-size:12px;padding:5px 10px;',
    onclick: () => { p.status = 'validated'; p.trainerNote = noteInput.value.trim(); saveState().then(refresh); }
  }, ['Valider']));
  controls.appendChild(el('button', {
    style:'border-color:#f09595;background:var(--danger-bg);color:var(--danger-text);font-weight:500;font-size:12px;padding:5px 10px;',
    onclick: () => { p.status = 'rejected'; p.trainerNote = noteInput.value.trim(); saveState().then(refresh); }
  }, ['À revoir']));
  controls.appendChild(el('button', {
    style:'font-size:12px;padding:5px 10px;',
    onclick: () => { p.trainerNote = noteInput.value.trim(); saveState().then(refresh); }
  }, ['Enregistrer commentaire']));
  row.appendChild(controls);
  return row;
}

loadState(() => {
  if (!STATE.apprentices) STATE.apprentices = [];

    if (!a.progress) a.progress = emptyProgressFor();
    RESOURCES.forEach(r => { if (!a.progress[r.id]) a.progress[r.id] = {status:'todo', note:'', trainerNote:'', updatedAt:null}; });
  });
  render();
});
